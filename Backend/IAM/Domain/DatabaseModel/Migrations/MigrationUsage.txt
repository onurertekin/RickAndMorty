# Adds a new migration. #
add-migration Initial -OutputDir "Migrations/MsSql" -Context MainDbContext           #Initial yerine o an hangi migration'ı yapıcaksan Mesela MusicTypesTableAdded yazabilirsin Initial yerine 

# Updates the database to the last migration or to a specified migration. #
update-database -Context MainDbContext

# Rollback migration. #
update-database -Context MainDbContext 0

# Remove migration file from project #
remove-migration -Context MainDbContext
